    Middleware

    Protecting Routes with net/http

    
func AuthMiddleware(engine *cryden.Engine) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // Get token from Authorization header
            authHeader := r.Header.Get("Authorization")
            if authHeader == "" {
                http.Error(w, "unauthorized", http.StatusUnauthorized)
                return
            }
            
            // Extract Bearer token
            tokenString := strings.TrimPrefix(authHeader, "Bearer ")
            
            // Verify token
            userID, err := cryden.VerifyToken(engine, tokenString)
            if err != nil {
                http.Error(w, "invalid token", http.StatusUnauthorized)
                return
            }
            
            // Add user ID to context
            ctx := context.WithValue(r.Context(), "user_id", userID)
            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}

// Usage
mux := http.NewServeMux()
mux.Handle("/protected", AuthMiddleware(engine)(http.HandlerFunc(protectedHandler)))

    Custom Middleware

    
func RateLimitHeaders(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        // Get rate limit result from context (set by Cryden)
        if result, ok := r.Context().Value("rate_limit").(*cryden.LimitResult); ok {
            w.Header().Set("X-RateLimit-Limit", strconv.Itoa(result.Limit))
            w.Header().Set("X-RateLimit-Remaining", strconv.Itoa(result.Remaining))
            w.Header().Set("X-RateLimit-Reset", strconv.FormatInt(int64(result.Reset.Seconds()), 10))
        }
        next.ServeHTTP(w, r)
    })
}
