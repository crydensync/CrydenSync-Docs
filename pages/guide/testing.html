    Testing

    Cryden is designed to be extremely testable. Here's how:

    In-Memory Storage (No Cleanup)

    
                    func TestLogin(t *testing.T) {
                        // Use in-memory stores - no cleanup needed!
                        engine := cryden.New()
                        ctx := context.Background()
                    
                        // Create test user
                        cryden.SignUp(ctx, engine, "test@example.com", "password123")
                    
                        // Test login
                        tokens, _, err := cryden.Login(ctx, engine, "test@example.com", "password123")
                        if err != nil {
                            t.Errorf("Login failed: %v", err)
                        }
                    
                        if tokens.AccessToken == "" {
                            t.Error("Expected access token")
                        }
                    }

    Mock Hasher (Fast Tests)

    
                    func init() {
                        // Use mock hasher for all tests (super fast!)
                        engine := cryden.New()
                        engine.WithHasher(&core.MockHasher{})
                    }

    No-Op Rate Limiter

    
                    func TestWithoutRateLimiting(t *testing.T) {
                        engine := cryden.New()
                        engine.WithRateLimiter(&core.NoopRateLimiter{}) // Never blocks
                    
                        // Make 100 login attempts - all allowed
                        for i := 0; i < 100; i++ {
                            _, _, err := cryden.Login(ctx, engine, "test@example.com", "password123")
                            if err != nil {
                                t.Errorf("Attempt %d failed: %v", i, err)
                            }
                        }
                    }

    Deterministic Testing

    
func TestSpecificScenario(t *testing.T) {
    engine := cryden.New()

    // Create exactly the state you need
    user, _ := cryden.SignUp(ctx, engine, "alice@example.com", "pass123")
    tokens, _, _ := cryden.Login(ctx, engine, "alice@example.com", "pass123")

    // Test logout
    err := cryden.Logout(ctx, engine, tokens.RefreshToken)
    if err != nil {
        t.Errorf("Logout failed: %v", err)
    }
}

    Testing Database Implementations

    
func TestWithSQLite(t *testing.T) {
    // Use temporary file
    dbPath := filepath.Join(t.TempDir(), "test.db")
    engine, err := cryden.WithSQLite(dbPath)
    if err != nil {
        t.Fatal(err)
    }
    defer engine.Close()

    // Test with real SQLite
    // File is automatically cleaned up by t.TempDir()
}
