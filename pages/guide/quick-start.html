    Quick Start Guide

    Minimal Working Setup

    1. Create a new project

    
mkdir myapp
cd myapp
go mod init myapp
go get github.com/crydensync/cryden

    2. Create main.go

    
package main

import (
    "context"
    "fmt"
    "log"

    "github.com/crydensync/cryden"
)

func main() {
    ctx := context.Background()
    engine := cryden.New()

    // Sign up
    user, err := cryden.SignUp(ctx, engine, "demo@example.com", "DemoPass123")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("✅ User: %s\n", user.Email)

    // Login
    tokens, _, err := cryden.Login(ctx, engine, "demo@example.com", "DemoPass123")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("✅ Access token: %.30s...\n", tokens.AccessToken)
}

    3. Run

    
go run main.go
