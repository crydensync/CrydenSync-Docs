    Configuration

    Engine Options

    
  engine := cryden.New()  // In-memory default
  
  // With SQLite persistence
  engine, err := cryden.WithSQLite("users.db")
  
  // With custom JWT secret
  cryden.WithJWTSecret(engine, "your-256-bit-secret")
  
  // With custom rate limiter
  engine.WithRateLimiter(myCustomLimiter)

    Environment Variables

    
  # JWT secret (REQUIRED in production)
  export CRYDEN_JWT_SECRET="your-super-secret-key"
  
  # Database connections
  export DATABASE_URL="postgres://user:pass@localhost/db"
  export MONGODB_URI="mongodb+srv://user:pass@cluster.mongodb.net"

    Production Configuration

    
  engine := cryden.New()
  cryden.WithJWTSecret(engine, os.Getenv("CRYDEN_JWT_SECRET"))
  
  // Use PostgreSQL
  store, _ := postgres.NewUserStore(os.Getenv("DATABASE_URL"))
  engine = engine.WithUserStore(store)  // Coming soon

    Default Values

    Setting Default Description
    Password min length 8 Security baseline
    Access token TTL 15m Short-lived JWT
    Refresh token TTL 7d Long-lived session
    Rate limit 5 per minute Per IP
    Hasher bcrypt (cost 10) Password hashing
