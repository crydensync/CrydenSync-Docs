    Auth API Reference

    Core authentication functions for CrydenSync engine.

    Sign Up

    
func SignUp(
    ctx context.Context,
    engine *Engine,
    email, password string,
) (*User, error)

    Returns: User on success

    Errors: ErrInvalidEmail, ErrPasswordTooShort, ErrUserExists

    Login

    
func Login(
    ctx context.Context,
    engine *Engine,
    email, password string,
) (*TokenPair, *LimitResult, error)

    Returns: TokenPair + rate limit info

    Errors: ErrInvalidCredentials, ErrTooManyAttempts

    Logout

    
func Logout(
    ctx context.Context,
    engine *Engine,
    refreshToken string,
) error

    Errors: ErrInvalidToken

func LogoutAll(

    Types

    TokenPair

    
type TokenPair struct {
    AccessToken  string
    RefreshToken string
    TokenType    string // "Bearer"
    ExpiresIn    int64  // seconds
}

    LimitResult

    
type LimitResult struct {
    Allowed   bool
    Limit     int
    Remaining int
    Reset     time.Duration
}
