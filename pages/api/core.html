    Core API Reference

    Core engine configuration and low-level primitives for CrydenSync.

    Engine

    
type Engine struct { /* private fields */ }

// New creates an in-memory engine (for testing)
func New() *Engine

// WithSQLite creates an engine with SQLite persistence
func WithSQLite(dbPath string) (*Engine, error)

// WithJWTSecret sets a custom JWT secret
func WithJWTSecret(engine *Engine, secret string) *Engine

    Configuration Methods

    
func (e *Engine) WithHasher(hasher Hasher) *Engine

    Token Verification

    
func VerifyToken(engine *Engine, tokenString string) (string, error)
// Returns userID or error

    Example

    
engine := cryden.New()

cryden.WithJWTSecret(engine, "your-secret")

userID, err := cryden.VerifyToken(engine, tokenString)
if err != nil {
    // Handle invalid token
}
